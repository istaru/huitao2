获取验证码接口:

	接口参数		:	{
						"phone":"13482509858",					//手机号
						"type":"1"								//验证码类型:1.注册 2.改密
	}
	返回参数		:	{
					"msg":"验证码已发送",       					//或相关错误信息
					"status":1           	 					//1成功,-1失败
	}

淘宝授权

	接口名     : /TaoBaoAuth/authInfo
	接口参数   : {
		"user_id":""            //用户uid[必传]
		"taobao_id":""          //淘宝openId[必传]
		"user_name":""          //淘宝昵称[必传]
		"user_head_img":""      //淘宝头像[必传]
		"imei":""               //安卓设备id[特殊可选]
		"bdid":""               //IOS bdid[特殊可选]
		"uuid":""               //IOS uuid [特殊可选]
		"idfa":""               //IOS idfa[特殊可选]
	}
	返回参数 : {
		"status": 1,     // or -1
		"msg": "授权成功" // OR 授权失败,
		"data": []
	}
